********** MODEL NAME

modelLV

********** MODEL NOTES

Simple Lotka Volterra

********** MODEL STATES
d/dt(x1) = -(a21 + a31 + a01) * x1 + a12 * x2 + a13 * x3
d/dt(x2) = a21 * x1 - a12 * x2
d/dt(x3) = a31 * x1 - a13 * x3

********** MODEL PARAMETERS

a12 = 0.3
a13 = 0.2
a21 = 0.5
a31 = 0.6
a01 = -0.2

********** MODEL VARIABLES

OUTPUT1 = x1
OUTPUT2 = x2
